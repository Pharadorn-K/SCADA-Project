// backend/node/services/persistenceEngine.js
const stateStore = require('./stateStore');
const { getDbPool } = require('./db');

async function saveAllShifts() {
  const pool     = await getDbPool();
  const machines = stateStore.getPlcSnapshot().machines;

  for (const [key, machine] of Object.entries(machines)) {
    if (!machine.shiftDurations) {
      machine.shiftDurations = {
        run_seconds: 0, idle_seconds: 0,
        alarm_seconds: 0, offline_seconds: 0
      };
    }
    if (!machine.shiftDate || !machine.shift) continue;

    const d            = machine.shiftDurations;
    const planned      = d.run_seconds + d.idle_seconds + d.alarm_seconds + d.offline_seconds;
    const avail        = planned > 0 ? d.run_seconds / planned : null;
    const count_output = machine.tags?.count_shift ?? 0;

    await pool.query(
      `INSERT INTO machine_shift_summary
         (date, shift, department, machine,
          run_seconds, idle_seconds, alarm_seconds, offline_seconds,
          availability, count_output)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
       ON DUPLICATE KEY UPDATE
         run_seconds     = VALUES(run_seconds),
         idle_seconds    = VALUES(idle_seconds),
         alarm_seconds   = VALUES(alarm_seconds),
         offline_seconds = VALUES(offline_seconds),
         availability    = VALUES(availability),
         count_output    = VALUES(count_output),
         updated_at      = NOW()`,
      [
        machine.shiftDate, machine.shift,
        machine.department, machine.machine,
        d.run_seconds, d.idle_seconds, d.alarm_seconds, d.offline_seconds,
        avail, count_output
      ]
    );
  }
}

function startAutoSave() {
  setInterval(saveAllShifts, 60 * 1000);
}

async function saveMachineShift(machine) {
  const pool         = await getDbPool();
  const d            = machine.shiftDurations;
  const planned      = d.run_seconds + d.idle_seconds + d.alarm_seconds + d.offline_seconds;
  const avail        = machine.availability ?? (planned > 0 ? d.run_seconds / planned : 0);
  const count_output = machine.tags?.count_shift ?? 0;

  await pool.query(
    `INSERT INTO machine_shift_summary
       (date, shift, department, machine,
        run_seconds, idle_seconds, alarm_seconds, offline_seconds,
        availability, count_output)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
     ON DUPLICATE KEY UPDATE
       run_seconds     = VALUES(run_seconds),
       idle_seconds    = VALUES(idle_seconds),
       alarm_seconds   = VALUES(alarm_seconds),
       offline_seconds = VALUES(offline_seconds),
       availability    = VALUES(availability),
       count_output    = VALUES(count_output),
       updated_at      = NOW()`,
    [
      machine.shiftDate,
      machine.shift,
      machine.department,
      machine.machine,
      d.run_seconds,
      d.idle_seconds,
      d.alarm_seconds,
      d.offline_seconds,
      avail,
      count_output
    ]
  );
}

function startDurationTicker() {
  setInterval(() => {
    const machines = stateStore.getPlcSnapshot().machines;
    for (const machine of Object.values(machines)) {
      if (!machine.status || !machine.shiftDurations) continue;
      const bucketMap = {
        RUNNING: 'run_seconds',
        IDLE:    'idle_seconds',
        ALARM:   'alarm_seconds',
        OFFLINE: 'offline_seconds'
      };
      const bucket = bucketMap[machine.status];
      if (bucket) machine.shiftDurations[bucket] += 1;
    }
  }, 1000);
}

module.exports = {
  saveMachineShift,
  startAutoSave,
  startDurationTicker
};
