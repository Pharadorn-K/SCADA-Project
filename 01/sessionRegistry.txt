// backend/node/services/sessionRegistry.js
const SESSION_TIMEOUT_MS = 24 * 60 * 60 * 1000; // match cookie maxAge
const IDLE_WARN_MS = 8 * 60 * 60 * 1000;        // warn after 8 hrs idle

const sessions = new Map(); // sessionId → { userId, role, loginAt, lastSeenAt }

function register(sessionId, userId, role) {
  sessions.set(sessionId, {
    sessionId,
    userId,
    role,
    loginAt:    Date.now(),
    lastSeenAt: Date.now()
  });
  broadcast();
}

function touch(sessionId) {
  const s = sessions.get(sessionId);
  if (s) {
    s.lastSeenAt = Date.now();
  }
}

function remove(sessionId) {
  sessions.delete(sessionId);
  broadcast();
}

function getAll() {
  return Array.from(sessions.values());
}

// prune sessions that have exceeded cookie maxAge
function pruneExpired() {
  const now = Date.now();
  let changed = false;
  for (const [id, s] of sessions) {
    if (now - s.loginAt > SESSION_TIMEOUT_MS) {
      sessions.delete(id);
      changed = true;
      global.services?.logService?.log({
        type: 'AUDIT',
        severity: 'WARN',
        user: s.userId,
        role: s.role,
        action: 'SESSION_TIMEOUT',
        message: `Session expired after ${Math.round((now - s.loginAt) / 3600000)}h`
      });
    }
  }
  if (changed) broadcast();
}

function broadcast() {
  const wss = global.services?.wss;
  if (!wss) return;
  const payload = JSON.stringify({
    type: 'session_update',
    sessions: getAll()
  });
  wss.clients.forEach(c => {
    if (c.readyState === 1) c.send(payload);
  });
}

// prune every minute
setInterval(pruneExpired, 60 * 1000);

module.exports = { register, touch, remove, getAll, IDLE_WARN_MS };
