// backend/node/services/shiftEngine.js
const stateStore        = require('./stateStore');
const persistenceEngine = require('./persistenceEngine');

// ── Shift boundary definitions ──────────────────────────────────────────
// Returns { shift, date } where date is always the START DATE of that shift.
//
// KEY FIX: Build the date string from local-time components instead of
// toISOString() which always returns UTC. On a UTC+7 server, 01:00 local
// is 18:00 UTC the previous day, so toISOString() would give yesterday's
// date BEFORE the -1 adjustment fires, causing a double-subtraction bug.
function getShiftInfo(timestamp = Date.now()) {
  const d    = new Date(timestamp);
  const hour = d.getHours(); // local hour

  let shift;
  let shiftDate = new Date(d);

  if (hour >= 6 && hour < 14) {
    shift = 'A';
  } else if (hour >= 14 && hour < 22) {
    shift = 'B';
  } else {
    shift = 'C';
    // Shift C started the previous day if we're between midnight and 06:00
    if (hour < 6) {
      shiftDate.setDate(shiftDate.getDate() - 1);
    }
  }

  // Build date string from LOCAL time components — not toISOString() (UTC)
  const pad = n => String(n).padStart(2, '0');
  const date = `${shiftDate.getFullYear()}-${pad(shiftDate.getMonth() + 1)}-${pad(shiftDate.getDate())}`;

  return { shift, date };
}

// Returns the next shift boundary as a Date (local time)
// Fires 5 seconds PAST the boundary to avoid timer-drift false positives
// (setTimeout can fire a few ms early; getShiftInfo would still say old shift)
function getNextShiftBoundary(now = new Date()) {
  const hour = now.getHours();
  const next = new Date(now);

  if (hour < 6) {
    next.setHours(6, 0, 5, 0);   // 06:00:05
  } else if (hour < 14) {
    next.setHours(14, 0, 5, 0);  // 14:00:05
  } else if (hour < 22) {
    next.setHours(22, 0, 5, 0);  // 22:00:05
  } else {
    // Past 22:00 — next boundary is 06:00 tomorrow
    next.setDate(next.getDate() + 1);
    next.setHours(6, 0, 5, 0);
  }

  return next;
}

function calculateAvailability(d) {
  const planned = d.run_seconds + d.idle_seconds + d.alarm_seconds + d.offline_seconds;
  return planned === 0 ? 0 : d.run_seconds / planned;
}

// ── Close old shift (save final availability) ───────────────────────────
async function closeShiftForMachine(machine) {
  const availability = calculateAvailability(machine.shiftDurations);
  try {
    await persistenceEngine.saveMachineShift({ ...machine, availability });
  } catch (e) {
    console.error('closeShiftForMachine error:', e.message);
  }
}

// ── Open new shift ─────────────────────────────────────────────────────
async function openNewShiftForMachine(machine, now) {
  const shiftInfo = getShiftInfo(now);
  machine.shift          = shiftInfo.shift;
  machine.shiftDate      = shiftInfo.date;
  machine.shiftDurations = { run_seconds: 0, idle_seconds: 0, alarm_seconds: 0, offline_seconds: 0 };
  machine.statusStartedAt = now;

  // ── Reset PLC count tag so Performance doesn't use old shift's count ──
  if (machine.tags) {
    machine.tags.count_shift = 0;
  }

  try {
    await persistenceEngine.saveMachineShift({ ...machine, availability: 0 });
  } catch (e) {
    console.error('openNewShiftForMachine error:', e.message);
  }
}
// ── Shift boundary: close all + open all ────────────────────────────────
async function processShiftBoundary() {
  console.log('🔄 SHIFT BOUNDARY TRIGGERED');
  const machines = stateStore.getPlcSnapshot().machines;
  const now      = Date.now();

  for (const machine of Object.values(machines)) {
    await closeShiftForMachine(machine);
    await openNewShiftForMachine(machine, now);
  }

  const next = getNextShiftBoundary();
  console.log('✅ New shift initialized');

  global.services?.plcMonitor?.broadcast({
    type: 'plc_snapshot',
    payload: stateStore.getPlcSnapshot()
  });
}

// ── Boundary scheduler ──────────────────────────────────────────────────
// Fires 5 seconds past each boundary so getShiftInfo() reliably returns
// the new shift even if setTimeout fires slightly early.
function scheduleShiftBoundary() {
  const now   = new Date();
  const next  = getNextShiftBoundary(now);
  const delay = next.getTime() - now.getTime();

  console.log('⏳ Next shift boundary at:',next.toLocaleString('th-TH', { timeZone: 'Asia/Bangkok' }));

  setTimeout(async () => {
    await processShiftBoundary();
    scheduleShiftBoundary(); // reschedule for the next boundary
  }, delay);
}

// ── Called once on bootstrap ────────────────────────────────────────────
// Checks whether the server was down during a shift change
async function checkMissedBoundary() {
  const machines  = stateStore.getPlcSnapshot().machines;
  const shiftInfo = getShiftInfo(Date.now());
  let anyMismatch = false;

  for (const machine of Object.values(machines)) {
    if (machine.shift !== shiftInfo.shift || machine.shiftDate !== shiftInfo.date) {
      anyMismatch = true;
      break;
    }
  }

  if (anyMismatch) {
    console.log('⚠️  Missed shift boundary detected on startup — processing now');
    await processShiftBoundary();
  } else {
    console.log('✅ Shift boundary check passed — current shift matches DB');
  }
}

// ── detectAndHandleShift: called on each PLC event ─────────────────────
// Guard prevents concurrent handling for the same machine
const _shiftChangeInProgress = new Set();

async function detectAndHandleShift(key) {
  if (_shiftChangeInProgress.has(key)) return;

  const machine   = stateStore.getPlc(key);
  if (!machine) return;

  const shiftInfo = getShiftInfo(Date.now());

  // No change → nothing to do
  if (machine.shift === shiftInfo.shift && machine.shiftDate === shiftInfo.date) return;

  _shiftChangeInProgress.add(key);
  try {
    console.log(`🔄 Shift change for ${key}: ${machine.shiftDate}/${machine.shift} → ${shiftInfo.date}/${shiftInfo.shift}`);
    await closeShiftForMachine(machine);
    await openNewShiftForMachine(machine, Date.now());
  } finally {
    _shiftChangeInProgress.delete(key);
  }
}

// ── ensureShiftRowsExist: safety net called every 60s ──────────────────
// Catches machines that somehow missed a shift boundary
async function ensureShiftRowsExist() {
  const machines  = stateStore.getPlcSnapshot().machines;
  const shiftInfo = getShiftInfo(Date.now());

  for (const machine of Object.values(machines)) {
    if (!machine.shiftDate || !machine.shift) continue;
    if (machine.shift === shiftInfo.shift && machine.shiftDate === shiftInfo.date) continue;

    const key = `${machine.department}_${machine.machine}`;
    if (_shiftChangeInProgress.has(key)) continue;

    _shiftChangeInProgress.add(key);
    try {
      console.log(`⚠️  Late shift open for ${key}: was ${machine.shiftDate}/${machine.shift}, now ${shiftInfo.date}/${shiftInfo.shift}`);
      await closeShiftForMachine(machine);
      await openNewShiftForMachine(machine, Date.now());
    } finally {
      _shiftChangeInProgress.delete(key);
    }
  }
}

module.exports = {
  detectAndHandleShift,
  scheduleShiftBoundary,
  checkMissedBoundary,
  ensureShiftRowsExist,
  getShiftInfo,
  getNextShiftBoundary
};
